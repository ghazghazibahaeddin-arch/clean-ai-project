async function sendMessage() { const input = document.getElementById('user-input'); const status = document.getElementById('status'); const prompt = input.value; if(!prompt) return; status.innerText = "⏳ Processing Securely..."; try { const response = await fetch('/api/chat', { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify({ prompt: prompt }) }); const data = await response.json(); if (response.ok) { addMessageToChat('AI', data.reply); status.innerText = "✅ Secure Connection Active"; } else { status.innerText = "❌ Error: " + (data.error || "Server Issue"); } } catch (e) { status.innerText = "❌ Fatal: API Connection Lost."; } }
